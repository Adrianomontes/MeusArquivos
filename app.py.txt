def buscar_ultimo_faturamento():

    padrao_busca = os.path.join(pasta_downloads, "NF-e_emitidas_endereços*.xlsx")
    arquivos_encontrados = glob.glob(padrao_busca)

def inicializar_banco():
    conn = sqlite3.connect(DB_FILE)
    cursor = conn.cursor()
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS ocorrencias (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            cliente TEXT, nota_fiscal TEXT, data TEXT, tratativa TEXT, motivo TEXT
        )
    ''')
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS coletas (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            cliente TEXT, nota_fiscal TEXT, transportadora TEXT, data_solicitacao TEXT, status TEXT
        )
    ''')
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS transportadoras (
            id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT UNIQUE, telefone TEXT, responsavel TEXT
        )
    ''')
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS canhotos (
            id INTEGER PRIMARY KEY AUTOINCREMENT, cliente TEXT, nota_fiscal TEXT, data_recebimento TEXT, status TEXT, observacoes TEXT
        )
    ''')
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS entregas_efetuadas (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            nota_fiscal TEXT UNIQUE,
            data_entrega TEXT
        )
    ''')
    cursor.execute("SELECT COUNT(*) FROM transportadoras")
    if cursor.fetchone() == 0:
        padroes = [('Braspress', '', ''), ('TNT Mercurio', '', ''), ('Jamef', '', ''), ('Planalto', '', '')]
        cursor.executemany("INSERT INTO transportadoras (name, telefone, responsavel) VALUES (?, ?, ?)", padroes)
    conn.commit()
    conn.close()

def ler_lista_do_excel():
    itens = []
    if os.path.exists(EXCEL_LISTA_MOTIVOS):
        try:
            df_motivos = pd.read_excel(EXCEL_LISTA_MOTIVOS)
            if df_motivos.shape >= 2:
                col_codigo = df_motivos.iloc[:, 0].fillna("").astype(str)
                col_motivo = df_motivos.iloc[:, 1].fillna("").astype(str)
                for cod, mot in zip(col_codigo, col_motivo):
                    if cod.strip() and mot.strip(): itens.append(f"{cod.strip()} - {mot.strip()}")
                    elif mot.strip(): itens.append(mot.strip())
        except: pass
    return itens

def ler_transportadoras_db():
    inicializar_banco()
    conn = sqlite3.connect(DB_FILE)
    cursor = conn.cursor()
    cursor.execute("SELECT name FROM transportadoras ORDER BY name ASC")
    rows = cursor.fetchall()
    conn.close()
    return [r for r in rows]

@app.route('/sistema_canhotos', methods=['GET'])

    filtro_cliente = request.args.get('busca_cliente', '').strip()
    conn = sqlite3.connect(DB_FILE)
    conn.row_factory = sqlite3.Row
    cursor = conn.cursor()
    if filtro_cliente:
        cursor.execute("SELECT * FROM canhotos WHERE LOWER(cliente) LIKE ? ORDER BY id DESC", (f'%{filtro_cliente.lower()}%',))

@app.route('/salvar', methods=['POST'])
def salvar():
    cliente = request.form['cliente']
    nf = request.form['nota_fiscal']
    data_orig = request.form['data']
    tratativa = request.form['tratativa']
    motivo = request.form['item_word']
    try: data_f = datetime.strptime(data_orig, '%Y-%m-%d').strftime('%d/%m/%Y')
    except: data_f = data_orig
    conn = sqlite3.connect(DB_FILE)
    cursor = conn.cursor()
    cursor.execute("INSERT INTO ocorrencias (cliente, nota_fiscal, data, tratativa, motivo) VALUES (?, ?, ?, ?, ?)", (cliente, nf, data_f, tratativa, motivo))
    conn.commit()
    conn.close()
    texto_gerado = f"Prezados,\n\nNova ocorrência da área de {tratativa}.\n\n• Cliente: {cliente}\n• NF: {nf}\n• Data: {data_f}\n• Motivo: {motivo}"
    return render_template('sucesso.html', texto_notificacao=texto_gerado)

@app.route('/relatorio_faturamento', methods=['GET'])
def relatorio_faturamento():
    hoje_str = datetime.now().strftime('%Y-%m-%d')
    filtro_inicio = request.args.get('data_inicio', '').strip()
    filtro_fim = request.args.get('data_fim', '').strip()
    filtro_nf = request.args.get('busca_nf', '').strip()
    filtro_cliente = request.args.get('busca_cliente', '').strip()
    filtro_transp = request.args.get('busca_transp', '').strip()
    busca_status = request.args.get('busca_status', '').strip()
    
    colunas, dados_lista, msg_erro = [], [], None
    total_faturadas, total_expedidas, total_em_aberto = 0, 0, 0
    
    excel_faturamento_atual = buscar_ultimo_faturamento()
    excel_expedicao_atual = buscar_ultima_expedicao()
    
    if excel_faturamento_atual and os.path.exists(excel_faturamento_atual):
        try:
            df = pd.read_excel(excel_faturamento_atual)
            col_data_real = 'Emissão da NF-e'
            col_nf_real = 'Número da NF-e'
            col_cli_real = 'Cliente'
            
            if col_data_real in df.columns:
                df[col_data_real] = df[col_data_real].replace(r'^\s*$', pd.NA, regex=True).ffill()
                df[col_data_real] = pd.to_datetime(df[col_data_real], errors='coerce')
                if filtro_inicio and filtro_fim:
                    df = df[(df[col_data_real] >= pd.to_datetime(filtro_inicio).normalize()) & (df[col_data_real] <= pd.to_datetime(filtro_fim).replace(hour=23,minute=59,second=59))]
                df[col_data_real] = df[col_data_real].dt.strftime('%d/%m/%Y')
                
            if col_nf_real in df.columns: 
                df[col_nf_real] = df[col_nf_real].astype(str).str.strip().str.replace(r'\.0$', '', regex=True)
            if col_cli_real in df.columns and filtro_cliente: 
                df = df[df[col_cli_real].astype(str).str.contains(filtro_cliente, case=False, na=False)]

            conn = sqlite3.connect(DB_FILE)
            df_db_entregas = pd.read_sql_query("SELECT nota_fiscal, data_entrega FROM entregas_efetuadas", conn)
            conn.close()
            df_db_entregas['nota_fiscal'] = df_db_entregas['nota_fiscal'].astype(str).str.strip().str.replace(r'\.0$', '', regex=True)

            df_exp_min = pd.DataFrame(columns=['Nota', 'Emissão', 'Transportadora'])
            if excel_expedicao_atual and os.path.exists(excel_expedicao_atual):
                df_exp = pd.read_excel(excel_expedicao_atual)
                col_dt_exp = 'Emissão'
                col_nf_exp = 'Nota'
                col_transp_exp = 'Transportadora'
                
                if col_nf_exp in df_exp.columns:
                    df_exp[col_nf_exp] = df_exp[col_nf_exp].astype(str).str.split(',')
                    df_exp = df_exp.explode(col_nf_exp)
                    df_exp[col_nf_exp] = df_exp[col_nf_exp].str.strip().str.replace(r'\.0$', '', regex=True)
                    df_exp = df_exp[df_exp[col_nf_exp] != ""]
                    
                if col_dt_exp in df_exp.columns:
                    df_exp[col_dt_exp] = df_exp[col_dt_exp].replace(r'^\s*$', pd.NA, regex=True).ffill()
                    df_exp[col_dt_exp] = pd.to_datetime(df_exp[col_dt_exp], errors='coerce').dt.strftime('%d/%m/%Y')

                cols_b = [col_nf_exp]
                if col_dt_exp in df_exp.columns: cols_b.append(col_dt_exp)
                if col_transp_exp in df_exp.columns: cols_b.append(col_transp_exp)
                df_exp_min = df_exp[cols_b].drop_duplicates(subset=[col_nf_exp])

            df = pd.merge(df, df_db_entregas, left_on=col_nf_real, right_on='nota_fiscal', how='left')
            if not df_exp_min.empty:
                df = pd.merge(df, df_exp_min, left_on=col_nf_real, right_on='Nota', how='left')

            status_lista, data_lista, transp_lista = [], [], []
            for _, r in df.iterrows():
                if pd.notna(r.get('data_entrega')) and str(r['data_entrega']).strip() != "":
                    status_lista.append('ENTREGUE')
                    data_lista.append(str(r['data_entrega']))
                    transp_lista.append(str(r.get('Transportadora_x', r.get('Transportadora', '---'))))
                elif 'Nota' in df.columns and pd.notna(r.get('Nota')):
                    status_lista.append('EXPEDIDO')
                    data_lista.append(str(r.get('Emissão', '---')))
                    transp_lista.append(str(r.get('Transportadora_y', '---')))
                else:
                    status_lista.append('NÃO EXPEDIDO')
                    data_lista.append('---')
                    transp_lista.append('---')

            df['STATUS EXPEDIÇÃO'] = status_lista
            df['DATA EXPEDIDO'] = data_lista
            df['TRANSP. EXPEDIÇÃO'] = transp_lista
            # 5. Aplica filtros secundários pós-cruzamento nas buscas
            if col_nf_real in df.columns and filtro_nf: 
                df = df[df[col_nf_real].astype(str).str.contains(filtro_nf, case=False, na=False)]
            if 'TRANSP. EXPEDIÇÃO' in df.columns and filtro_transp:
                df = df[df['TRANSP. EXPEDIÇÃO'].astype(str).str.contains(filtro_transp, case=False, na=False)]

            # 6. Cálculo dos Indicadores do Dashboard
            total_faturadas = len(df)
            total_expedidas = len(df[df['STATUS EXPEDIÇÃO'].isin(['EXPEDIDO', 'ENTREGUE'])])
            total_em_aberto = total_faturadas - total_expedidas

            # 7. Filtro Logístico de Status Selecionado
            if busca_status:
                df = df[df['STATUS EXPEDIÇÃO'] == busca_status]

            for col in df.columns:
                if df[col].dtype == 'datetime64[ns]': df[col] = df[col].dt.strftime('%d/%m/%Y')
            df = df.fillna("")

            # Filtra a lista de colunas para exibir apenas os dados limpos do Faturamento
            colunas = [c for c in df.columns if c not in ['STATUS EXPEDIÇÃO', 'DATA EXPEDIDO', 'TRANSP. EXPEDIÇÃO', 'Nota', 'Emissão', 'nota_fiscal', 'data_entrega'] and not c.endswith('_y') and not c.endswith('_x')]
            dados_lista = df.to_dict(orient='records')
        except Exception as e: msg_erro = f"Erro ao processar faturamento: {str(e)}"
    else: msg_erro = "Nenhum arquivo 'NF-e_emitidas_endereços' localizado em Downloads."
    
    return render_template('relatorio_faturamento.html', dados=dados_lista, colunas=colunas, filtro_nf=filtro_nf, filtro_cliente=filtro_cliente, filtro_transp=filtro_transp, busca_status=busca_status, data_inicio=filtro_inicio, data_fim=filtro_fim, total_faturadas=total_faturadas, total_expedidas=total_expedidas, total_em_aberto=total_em_aberto, erro=msg_erro)

@app.route('/relatorio', methods=['GET'])
def relatorio():
    filtro_cliente = request.args.get('busca_cliente', '').strip()
    filtro_nf = request.args.get('busca_nf', '').strip()
    conn = sqlite3.connect(DB_FILE)
    conn.row_factory = sqlite3.Row
    cursor = conn.cursor()
    query = "SELECT * FROM ocorrencias WHERE 1=1"
    params = []
    if filtro_cliente: query += " AND LOWER(cliente) LIKE ?"; params.append(f'%{filtro_cliente.lower()}%')
    if filtro_nf: query += " AND CAST(nota_fiscal AS TEXT) LIKE ?"; params.append(f'%{filtro_nf}%')
    query += " ORDER BY id DESC"
    cursor.execute(query, params)
    rows = cursor.fetchall()
    conn.close()
    lista = [{'Cliente': r['cliente'], 'Nota Fiscal': r['nota_fiscal'], 'Data': r['data'], 'Tratativa': r['tratativa'], 'Motivo': r['motivo']} for r in rows]
    return render_template('relatorio.html', ocorrencias=lista, filtro_cliente=filtro_cliente, filtro_nf=filtro_nf)

@app.route('/relatorio_coletas', methods=['GET'])
def relatorio_coletas():
    filtro_cliente = request.args.get('busca_cliente', '').strip()
    filtro_nf = request.args.get('busca_nf', '').strip()
    filtro_transp = request.args.get('busca_transp', '').strip()
    conn = sqlite3.connect(DB_FILE)
    conn.row_factory = sqlite3.Row
    cursor = conn.cursor()
    cursor.execute("SELECT COUNT(*) FROM coletas WHERE status LIKE '%Pendente%'")
    p_count = cursor.fetchone()
    cursor.execute("SELECT COUNT(*) FROM coletas WHERE status LIKE '%Efetuada%'")
    e_count = cursor.fetchone()
    query = "SELECT * FROM coletas WHERE 1=1"
    params = []
    if filtro_cliente: query += " AND LOWER(cliente) LIKE ?"; params.append(f'%{filtro_cliente.lower()}%')
    if filtro_nf: query += " AND CAST(nota_fiscal AS TEXT) LIKE ?"; params.append(f'%{filtro_nf}%')
    if filtro_transp: query += " AND LOWER(transportadora) LIKE ?"; params.append(f'%{filtro_transp.lower()}%')
    query += " ORDER BY id DESC"
    cursor.execute(query, params)
    rows = cursor.fetchall()
    conn.close()
    lista = [{'Cliente': r['cliente'], 'Nota Fiscal': r['nota_fiscal'], 'Transportadora': r['transportadora'], 'Data Solicitação': r['data_solicitacao'], 'Status': r['status']} for r in rows]
    return render_template('relatorio_coletas.html', coletas=lista, filtro_cliente=filtro_cliente, filtro_nf=filtro_nf, filtro_transp=filtro_transp, pendentes=p_count, efetuadas=e_count)

    query = "SELECT * FROM ocorrencias WHERE 1=1"

    conn.close()
